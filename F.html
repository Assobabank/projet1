function ajouterLigne() { // Obtenez une référence à la tbody du tableau var tbody = document.querySelector(".body"); // Récupérez les valeurs des champs d'entrée var nouveauNom = document.getElementById("nouveauNom").value; var nouvelleQuantite = document.getElementById("nouvelleQuantite").value; // Créez une nouvelle ligne avec les valeurs entrées var nouvelleLigne = document.createElement("tr"); nouvelleLigne.innerHTML = `
${nouveauNom} supprimer
${nouvelleQuantite}fr
${nouvelleQuantite}fr
`; // Ajoutez la nouvelle ligne à la tbody tbody.appendChild(nouvelleLigne); // Réinitialisez les champs d'entrée document.getElementById("nouveauNom").value = ""; document.getElementById("nouvelleQuantite").value = ""; } // Associez la fonction à l'événement du bouton "Ajouter un article" document.getElementById("ajouterLigne").addEventListener("click", ajouterLigne);
